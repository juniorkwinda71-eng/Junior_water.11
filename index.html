Quick Setup — Connect to Google Sheets (Free)
Responses will be saved to YOUR Google Sheet and shared across every device. Takes 3 minutes to set up once.
1
Go to script.google.com → click New Project
2
Delete all existing code → paste the Google Apps Script code shown below → click Save (floppy disk icon)
3
Click Deploy → New Deployment → Web App → set "Who has access" to Anyone → click Deploy → copy the Web App URL
4
Paste the Web App URL below and click Connect
— Paste this into Google Apps Script —
var SHEET_NAME='Responses';
function doGet(e){
  var ss=SpreadsheetApp.openById(e.parameter.sheetId||PropertiesService.getScriptProperties().getProperty('sheetId'));
  var sh=ss.getSheetByName(SHEET_NAME)||ss.insertSheet(SHEET_NAME);
  var action=e.parameter.action;
  if(action==='read'){var d=sh.getDataRange().getValues();return ok(d);}
  if(action==='delete'){var id=Number(e.parameter.id);var rows=sh.getDataRange().getValues();for(var i=rows.length-1;i>=1;i--){if(rows[i][0]==id){sh.deleteRow(i+1);break;}}return ok('deleted');}
  if(action==='clear'){if(sh.getLastRow()>1)sh.deleteRows(2,sh.getLastRow()-1);return ok('cleared');}
  return ok('ready');
}
function doPost(e){
  var data=JSON.parse(e.postData.contents);
  var ss=SpreadsheetApp.openById(data.sheetId);
  var sh=ss.getSheetByName(SHEET_NAME)||ss.insertSheet(SHEET_NAME);
  if(sh.getLastRow()===0){sh.appendRow(['ID','FirstName','LastName','FullName','Gender','Age','Occupation','Province','Email','Phone','Q1_Crisis','Q2_Shortages','Q3_Concern','Q4_Household','Q5_Methods','SavedAt']);}
  sh.appendRow([data.id,data.fn,data.ln,data.fullName,data.gender,data.age,data.occ,data.prov,data.email,data.phone,data.q1,data.q2,data.q3,data.q4,(data.q5||[]).join(', '),data.savedAt]);
  return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
}
function ok(d){return ContentService.createTextOutput(JSON.stringify({ok:true,data:d})).setMimeType(ContentService.MimeType.JSON);}
